192  * - if 2 number arguments given creates size with x = y = 0 and width and height set

194  * - if a Point and a Size given creates rect with point as an origin and given size

198  * @constructor

201     if (a3 !== undefined) {
202         this.x      = a1;
203         this.y      = a2;
204         this.width  = a3;
205         this.height = a4;

207         this.x      = 0;
208         this.y      = 0;
209         this.width  = a1 || 0;
210         this.height = a2 || 0;
211     } else {
212         this.x      = a1 ? a1.x      : 0;
213         this.y      = a1 ? a1.y      : 0;
214         this.width  = a2 ? a2.width  : 0;
215         this.height = a2 ? a2.height : 0;
216     }

442 Rect.fromCoords = function(minX, minY, maxX, maxY) {
443     if (maxX === undefined) {
444         return new Rect(
445             minX.x, 
446             minX.y, 
447             minY.x - minX.x, 
448             minY.y - minX.y
449         );
450     }
451     return new Rect(minX, minY, maxX - minX, maxY - minY);
452 };

650 function unitsToPx (units, relative) {
651     var m = (units + '').match(/([-0-9\.]+)(\S*)/),
652         v = parseFloat(m[1], 10),
653         u = (m[2] || '').toLowerCase();
654         
655     if (u) {
656         // if (u == 'px') v = v;
657         if (u == '%' && relative) v *= relative / 100;
658     }
659     if (v < 0 && relative) v = relative + v;
660     return v;
661 }
