 11     bind: function(name, callback) {
 12         uki.each(name.split(' '), function(i, name) {
 13             if (!this._bound(name)) this._bindToDom(name);
 14             this._observersFor(name).push(callback);
 15         }, this);
 16     },
 17     
 18     unbind: function(name, callback) {
 19         uki.each(name.split(' '), function(i, name) {
 20             this._observers[name] = uki.grep(this._observers[name], function(observer) {
 21                 return observer != callback;
 22             });
 23             if (this._observers[name].length == 0) {
 24                 this._unbindFromDom(name);
 25             }
 26         }, this);
 27     },
 28     
 29     trigger: function(name/*, data1, data2*/) {
 30         var attrs = Array[PROTOTYPE].slice.call(arguments, 1);
 31         uki.each(this._observersFor(name, true), function(i, callback) {
 32             callback.apply(this, attrs);
 33         }, this);
 34     },
 35     
 36     _unbindFromDom: function(name, target) {
 37         if (!this._domHander || !this._eventTargets[name]) return;
 38         uki.dom.unbind(this._eventTargets[name], name, this._domHander);
 39     },
 40     
 41     _bindToDom: function(name, target) {
 42         this._domHander = this._domHander || uki.proxy(function(e) {
 43             this.trigger(e.type, {domEvent: e, source: this});
 44         }, this);
 45         this._eventTargets = this._eventTargets || {};
 46         this._eventTargets[name] = target || this.dom();
 47         uki.dom.bind(this._eventTargets[name], name, this._domHander);
 48         return true;
 49     },
 50     
 51     _bound: function(name) {
 52         return this._observers && this._observers[name];
 53     },
 54     
 55     _observersFor: function(name, skipCreate) {
 56         if (skipCreate && (!this._observers || !this._observers[name])) return [];
 57         if (!this._observers) this._observers = {};
 58         if (!this._observers[name]) this._observers[name] = [];
 59         return this._observers[name];
 60     }
 61 };
