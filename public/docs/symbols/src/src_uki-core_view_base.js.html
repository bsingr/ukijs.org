 28     /**
 29      * Base class for all uki views.
 30      *
 31      * <p>View creates and layouts dom nodes. uki.view.Base defines basic API for other views.
 32      * It also defines common layout algorithms.</p>
 33      *
 34      * Layout
 35      *
 36      * <p>View layout is defined by rectangle and anchors.
 37      * Rectangle is passed to constructor, anchors are set through the #anchors attribute.</p>
 38      * 
 39      * <p>Rectangle defines initial position and size. Anchors specify how view will move and
 40      * resize when its parent is resized.</p>
 41      *
 42      * 2 phases of layout
 43      *
 44      * <p>Layout process has 2 phases. 
 45      * First views rectangles are recalculated. This may happen several times before dom 
 46      * is touched. This is done either explictly through #rect attribute or through
 47      * #parentResized callbacks. 
 48      * After rectangles are set #layout is called. This actually updates dom styles.</p>
 49      *
 50      * @example
 51      * uki({ view: 'Base', rect: '10 20 100 50', anchors: 'left top right' })
 52      * // Creates Base view with initial x = 10px, y = 20px, width = 100px, height = 50px.
 53      * // When parent resizes x, y and height will stay the same. Width will resize with parent.
 54      *
 55      *
 56      * @see uki.view.Base#anchors
 57      * @constructor
 58      *
 59      * @name uki.view.Base
 60      * @implements uki.view.Observable
 61      * @param {uki.geometry.Rect} rect initial position and size
 62      */
 63     proto.init = function(rect) {
 64         this._parentRect = this._rect = Rect.create(rect);
 65         this._setup();
 66         uki.initNativeLayout();
 67         this._createDom();
 68     };

166     uki.each(['background', 'shadow'], function(i, name) {
167         var field = '_' + name,
168             defaultAttr = 'default' + name.substr(0, 1).toUpperCase() + name.substr(1),
169             defaultField = '_' + defaultAttr;
170 
171         proto[name] = function(val) {
172             if (val === undefined && !this[field] && this[defaultAttr]) this[field] = this[defaultAttr]();
173             if (val === undefined) return this[field];
174             val = uki.background(val);
175             
176             if (val == this[field]) return this;
177             if (this[field]) this[field].detach(this);
178             val.attachTo(this);
179             
180             this[field] = val;
181             return this;
182         };
183         
184         proto[defaultAttr] = function() {
185             return this[defaultField] && uki.theme.background(this[defaultField]);
186         };
187     });

218     /**
219      * Sets or retrieves view's position and size.
220      *
221      * @param {string|uki.geometry.Rect} newRect
222      * @returns {uki.view.Base} self
223      */
224     proto.rect = function(newRect) {
225         if (newRect === undefined) return this._rect;
226 
227         newRect = Rect.create(newRect);
228         this._parentRect = newRect;
229         // if (newRect.eq(this._rect)) return false;
230         this._rect = this._normalizeRect(newRect);
231         this._needsLayout = this._needsLayout || layoutId++;
232         return this;
233     };

235     /**
236      * Set or get sides which the view should be attached to.
237      * When a view is attached to a side the distance between this side and views border
238      * will remain constant on resize. Anchor can be any combination of
239      * "top", "right", "bottom", "left", "width" and "height". 
240      * If you set both "right" and "left" than "width" is assumed.
241      *
242      * Anchors are stored as a bitmask. Though its easier to set them using strings
243      *
244      * @function
245      * @param {string|number} anchors
246      * @returns {number|uki.view.Base} anchors or self
247      */
248     proto.anchors = uki.newProp('_anchors', function(anchors) {
249         if (anchors.indexOf) {
250             var tmp = 0;
251             if (anchors.indexOf('right'  ) > -1) tmp |= ANCHOR_RIGHT; 
252             if (anchors.indexOf('bottom' ) > -1) tmp |= ANCHOR_BOTTOM;
253             if (anchors.indexOf('top'    ) > -1) tmp |= ANCHOR_TOP;   
254             if (anchors.indexOf('left'   ) > -1) tmp |= ANCHOR_LEFT;  
255             if (anchors.indexOf('width'  ) > -1 || (tmp & ANCHOR_LEFT && tmp & ANCHOR_RIGHT)) tmp |= ANCHOR_WIDTH;  
256             if (anchors.indexOf('height' ) > -1 || (tmp & ANCHOR_BOTTOM && tmp & ANCHOR_TOP)) tmp |= ANCHOR_HEIGHT;
257             anchors = tmp;
258         }
259         this._anchors = anchors;
260         this._styleH = anchors & ANCHOR_LEFT ? 'left' : 'right';
261         this._styleV = anchors & ANCHOR_TOP ? 'top' : 'bottom';
262     });
263     
264     /**
265      * Returns rectangle for child layout. Usualy equals to #rect. Though in some cases,
266      * client rectangle my differ from #rect. Example uki.view.ScrollPane.
267      *
268      * @param {uki.view.Base} child 
269      * @returns {uki.geometry.Rect}
270      */
271     proto.rectForChild = function(child) {
272         return this.rect();

275     /**
276      * Updates dom to match #rect property.
277      *
278      * Layout is designed to minimize dom writes. If view is anchored to the right then
279      * style.right is used, style.left for left anchor, etc. If browser supports this
280      * both style.right and style.left are used. Otherwise style.width will be updated
281      * manualy on each resize. 
282      *
283      * @see uki.dom.initNativeLayout
284      */
285     proto.layout = function() {
286         this._layoutDom(this._rect);
287         this._needsLayout = false;
288         this.trigger('layout', {rect: this._rect, source: this});
289         this._firstLayout = false;
290     };

292     proto.minSize = uki.newProp('_minSize', function(s) {
293         this._minSize = Size.create(s);
294         this.rect(this._parentRect);
295         if (this._minSize.width) this._dom.style.minWidth = this._minSize.width + 'px';
296         if (this._minSize.height) this._dom.style.minHeight = this._minSize.height + 'px';
297     });

299     /**
300      * Resizes view when parent changes size acording to anchors.
301      * Called from parent view. Usualy after parent's #rect is called.
302      *
303      * @param {uki.geometry.Rect} oldSize
304      * @param {uki.geometry.Rect} newSize
305      */
306     proto.parentResized = function(oldSize, newSize) {
307         var newRect = this._parentRect.clone(),
308             dX = (newSize.width - oldSize.width) /
309                 ((this._anchors & ANCHOR_LEFT ^ ANCHOR_LEFT ? 1 : 0) +   // flexible left
310                 (this._anchors & ANCHOR_WIDTH ? 1 : 0) +             
311                 (this._anchors & ANCHOR_RIGHT ^ ANCHOR_RIGHT ? 1 : 0)),   // flexible right
312             dY = (newSize.height - oldSize.height) /
313                 ((this._anchors & ANCHOR_TOP ^ ANCHOR_TOP ? 1 : 0) +      // flexible top
314                 (this._anchors & ANCHOR_HEIGHT ? 1 : 0) + 
315                 (this._anchors & ANCHOR_BOTTOM ^ ANCHOR_BOTTOM ? 1 : 0)); // flexible right
316                 
317         if (this._anchors & ANCHOR_LEFT ^ ANCHOR_LEFT) newRect.x += dX;
318         if (this._anchors & ANCHOR_WIDTH) newRect.width += dX;
319 
320         if (this._anchors & ANCHOR_TOP ^ ANCHOR_TOP) newRect.y += dY;
321         if (this._anchors & ANCHOR_HEIGHT) newRect.height += dY;
322         this.rect(newRect);
323     };

325     /**
326      * Resizes view to its contents. Contents size is determined by view.
327      * View can be resized by width, height or both. This is specified throght
328      * autosizeStr param.
329      * View will grow shrink acording to its #anchors.
330      *
331      * @param {autosizeStr} autosize 
332      * @returns {uki.view.Base} self
333      */
334     proto.resizeToContents = function(autosizeStr) {
335         var autosize = decodeAutosize(autosizeStr);
336         if (0 == autosize) return this;
337         
338         var oldRect = this.rect(),
339             newRect = this._calcRectOnContentResize(autosize);
340         // if (newRect.eq(oldRect)) return this;
341         // this.rect(newRect);
342         this._rect = this._parentRect = newRect;
343         this._needsLayout = true;
344         return this;
345     };
346     
347     /**
348      * Calculates view's contents size. Redefined in subclasses.
349      *
350      * @param {number} autosize Bitmask
351      * @returns {uki.geometry.Rect}
352      */
353     proto.contentsSize = function(autosize) {
354         return this.rect();
355     };

357     proto._normalizeRect = function(rect) {
358         if (this._minSize) {
359             rect = new Rect(rect.x, rect.y, MAX(this._minSize.width, rect.width), MAX(this._minSize.height, rect.height));
360         }
361         return rect;
362     };

366     function decodeAutosize (autosizeStr) {
367         if (!autosizeStr) return 0;
368         var autosize = 0;
369         if (autosizeStr.indexOf('width' ) > -1) autosize = autosize | ANCHOR_WIDTH;
370         if (autosizeStr.indexOf('height') > -1) autosize = autosize | ANCHOR_HEIGHT;
371         return autosize;
372     }

375     proto._initBackgrounds = function() {
376         if (this.background()) this.background().attachTo(this);
377         if (this.shadow()) this.shadow().attachTo(this);
378     };
379     
380     proto._calcRectOnContentResize = function(autosize) {
381         var newSize = this.contentsSize( autosize ),
382             oldSize = this.rect();
383 
384         if (newSize.eq(oldSize)) return oldSize; // nothing changed
385         
386         // calculate where to resize
387         var newRect = this.rect().clone(),
388             dX = newSize.width - oldSize.width,
389             dY = newSize.height - oldSize.height;
390     
391         if (autosize & ANCHOR_WIDTH) {
392             if (this._anchors & ANCHOR_LEFT ^ ANCHOR_LEFT && this._anchors & ANCHOR_RIGHT ^ ANCHOR_RIGHT) {
393                 newRect.x -= dX/2;
394             } else if (this._anchors & ANCHOR_LEFT ^ ANCHOR_LEFT) {
395                 newRect.x -= dX;
396             }
397             newRect.width += dX;
398         }
399         
400         if (autosize & ANCHOR_HEIGHT) {
401             if (this._anchors & ANCHOR_TOP ^ ANCHOR_TOP && this._anchors & ANCHOR_BOTTOM ^ ANCHOR_BOTTOM) {
402                 newRect.y -= dY/2;
403             } else if (this._anchors & ANCHOR_TOP ^ ANCHOR_TOP) {
404                 newRect.y -= dY;
405             }
406             newRect.height += dY;
407         }
408         
409         return newRect;
410     };

420     /**
421      * Called through a second layout pass when _dom should be created
422      */
423     proto._createDom = function() {
424         this._dom = uki.createElement('div', this.defaultCss);
425     };

427     /**
428      * Called through a second layout pass when _dom is allready created
429      */
430     proto._layoutDom = function(rect) {
431         var l = {}, s = uki.supportNativeLayout, relativeRect = this.parent().rectForChild(this);
432         if (s && this._anchors & ANCHOR_LEFT && this._anchors & ANCHOR_RIGHT) {
433             l.left = rect.x;
434             l.right = relativeRect.width - rect.x - rect.width;
435         } else {
436             l.width = rect.width;
437             l[this._styleH] = this._styleH == 'left' ? rect.x : relativeRect.width - rect.x - rect.width;
438         }
439         
440         if (s && this._anchors & ANCHOR_TOP && this._anchors & ANCHOR_BOTTOM) {
441             l.top = rect.y;
442             l.bottom = relativeRect.height - rect.y - rect.height;
443         } else {
444             l.height = rect.height;
445             l[this._styleV] = this._styleV == 'top'  ? rect.y : relativeRect.height - rect.y - rect.height;
446         }
447         this._lastLayout = uki.dom.layout(this._dom.style, l, this._lastLayout);
448         if (this._firstLayout) this._initBackgrounds();
449         return true;
450     };

452     proto._bindToDom = function(name) {
453         if ('resize layout'.indexOf(name) > -1) return true;
454         return uki.view.Observable._bindToDom.call(this, name);
