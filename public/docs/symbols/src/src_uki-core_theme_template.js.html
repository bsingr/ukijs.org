  3 /**
  4  * Simple and fast (2xâ€“15x faster than regexp) html template
  5  * @example
  6  *   var t = new uki.theme.Template('<p class="${className}">${value}</p>')
  7  *   t.render({className: 'myClass', value: 'some html'})
  8  */
  9 uki.theme.Template = function(code) {
 10     var parts = code.split('${'), i, l, tmp;
 11     this.parts = [parts[0]];
 12     this.names = [];
 13     for (i=1, l = parts.length; i < l; i++) {
 14         tmp = parts[i].split('}');
 15         this.names.push(tmp[0]);
 16         this.parts.push('');
 17         this.parts.push(tmp[1]);
 18     };
 19 };
 20 
 21 uki.theme.Template.prototype.render = function(values) {
 22     for (var i=0, names = this.names, l = names.length; i < l; i++) {
 23         this.parts[i*2+1] = values[names[i]] || '';
 24     };
 25     return this.parts.join('');
 26 };
