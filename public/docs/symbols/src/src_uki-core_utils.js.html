 12     /**
 13      * <p>Sets or retrieves attribute on an object. </p>
 14      * <p>If target has function with attr it will be called target[attr](value=)
 15      * If no function present attribute will be set/get directly: target[attr] = value or return target[attr]</p>
 16      *
 17      * @example
 18      *   uki.attr(view, 'name', 'funny') // sets name to funny on view
 19      *   uki.attr(view, 'id') // gets id attribute of view
 20      *
 21      * @param {object} target
 22      * @param {string} attr Attribute name
 23      * @param {object=} value Value to set
 24      * @returns {object} target if value is being set, retrieved value otherwise
 25      */
 26     attr: function(target, attr, value) {
 27         if (value !== undefined) {
 28             // if (target[attr].apply) {
 29             if (utils.isFunction(target[attr])) {
 30                 target[attr](value);
 31             } else {
 32                 target[attr] = value;
 33             }
 34             return target;
 35         } else {
 36             // if (target[attr].apply) {
 37             if (utils.isFunction(target[attr])) {
 38                 return target[attr]();
 39             } else {
 40                 return target[attr];
 41             }
 42         }
 43     },

104     /**
105      * Iterates through all non empty values of object or an Array
106      *
107      * @param {object|Array} object Object to iterate through
108      * @param {function(number, object):boolean} callback Called for every item, may return false to stop iteration
109      * @param {object} context Context in which callback should called. If not specified context will be set to
110      *                         current item
111      * @returns {object}
112      */
113     each: function( object, callback, context ) {
114         var name, i = 0, length = object.length;
115 
116         if ( length === undefined ) {
117             for ( name in object ) {
118                 if ( !name || object[ name ] === undefined || !object.hasOwnProperty(name) ) continue;
119                 if ( callback.call( context || object[ name ], name, object[ name ] ) === false ) { break; }
120             }
121         } else {
122             for ( var value = object[0]; i < length && callback.call( context || value, i, value ) !== false; value = object[++i] ){}
123         }
124         return object;
125     },

149         var ret = [], done = {};

151         try {
152 
153             for ( var i = 0, length = array.length; i < length; i++ ) {
154                 var id = array[ i ];
155 
156                 if ( !done[ id ] ) {
157                     done[ id ] = true;
158                     ret.push( array[ i ] );
159                 }
160             }
161 
162         } catch( e ) {

164         }

198         var ret = [],
199             mapper = utils.isFunction(callback) ? callback :

260     /**
261      * <p>Creates a new class inherited from base classes. Init function is used as constructor</p>
262      * @example
263      *   baseClass = uki.newClass({
264      *      init: function() { this.x = 3 }
265      *   });
266      *
267      *   childClass = uki.newClass(baseClass, {
268      *      getSqrt: function() { return this.x*this.x }
269      *   });
270      *
271      * @param {object=} superClass If superClass has prototype "real" protototype base inheritance is used,
272      *                             otherwise superClass properties ar simply copied to newClass prototype
273      * @param {Array.<object>=} mixins
274      * @param {object} methods
275      * @returns Describe what it returns
276      */
277     newClass: function(/* [[superClass], mixin1, mixin2, ..] */ methods) {
278         var klass = function() {
279                 this.init.apply(this, arguments);
280             };
281             
282         var inheritance, i, startFrom = 0;
283             
284         if (arguments.length > 1) {
285             if (arguments[0][PROTOTYPE]) { // real inheritance
286                 /** @ignore */
287                 inheritance = function() {};
288                 inheritance[PROTOTYPE] = arguments[0][PROTOTYPE];
289                 klass[PROTOTYPE] = new inheritance();
290                 startFrom = 1;
291             }
292         }
293         for (i=startFrom; i < arguments.length; i++) {
294             utils.extend(klass[PROTOTYPE], arguments[i]);
295         };
296         return klass;
297     },

299     /**
300      * <p>Creates default uki property function</p>
301      * <p>If value is given to this function it sets property to value
302      * If no arguments given than function returns current property value</p>
303      *
304      * <p>Optional setter can be given. In this case setter will be called instead
305      * of simple this[field] = value</p>
306      *
307      * <p>If used as setter function returns self</p>
308      *   
309      * @example
310      *   x.width = uki.newProperty('_width');
311      *   x.width(12); // x._width = 12
312      *   x.width();   // return 12
313      *
314      * @param {string} field Field name
315      * @param {function(object)=} setter
316      * @returns {function(object=):object}
317      */
318     newProp: function(field, setter) {
319         return function(value) {
320             if (value === undefined) return this[field];
321             if (setter) { setter.call(this, value); } else { this[field] = value; };
322             return this;
323         };
324     },

326     /**
327      * <p>Adds several properties (uki.newProp) to a given object.</p>
328      * <p>Field name equals to '_' + property name</p>
329      *
330      * @example
331      *   uki.addProps(x, ['width', 'height'])
332      *
333      * @param {object} proto Object to add properties to
334      * @param {Array.<string>} props Property names
335      */
336     addProps: function(proto, props) {
337         utils.each(props, function() { proto[this] = utils.newProp('_' + this); });
338     },
339     
340     delegateProp: function(proto, name, target) {
341         var propName = '_' + name;
342         proto[name] = function(value) {
343             if (this[target]) return utils.attr(this[target], name, value);
344             if (value === undefined) return this[propName];
345             this[propName] = value;
346             return this;
347         };
348     }
