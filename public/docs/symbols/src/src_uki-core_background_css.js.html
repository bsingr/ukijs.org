  4 /**
  5  * Writes css properties to targets dom()
  6  *
  7  * @class
  8  */
  9 uki.background.Css = uki.newClass(new function() {
 10     
 11     this.init = function(options) {
 12         this._options = typeof options == 'string' ? {background: options} : options;
 13     };
 14     
 15     this.attachTo = function(comp) {
 16         this._comp = comp;
 17         this._originalValues = {};
 18         
 19         uki.each(this._options, function(name, value) {
 20             // this._originalValues[name] = dom.style[name];
 21             // dom.style[name] = value;
 22             this._originalValues[name] = comp.style(name);
 23             comp.style(name, value);
 24         }, this);
 25     };
 26     
 27     this.detach = function() {
 28         if (this._comp) {
 29             uki.each(this._options, function(name, value) {
 30                 this._comp.style(name, this._originalValues[name]);
 31             }, this);
 32         }
 33         
 34     };
 35 });
