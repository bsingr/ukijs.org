  6 uki.view.Stylable = new function() {
  7     var proto = this;

  9     proto.style = function(name, value) {
 10         if (typeof name == 'string') return this._style(name, value);
 11         
 12         uki.each(name, function(name, value) {
 13             this._style(name, value);
 14         }, this);
 15         return this;
 16     };
 17     
 18     proto._style = function(name, value) {
 19         if (value === undefined) return this._dom.style[name];
 20         this._dom.style[name] = value;
 21         return this;
 22     };

 31     var probe = uki.createElement('div').style;
 32     uki.each(['userSelect', 'MozUserSelect', 'WebkitUserSelect'], function() {
 33         if (typeof probe[this] == 'string') proto._textSelectProp = this;
 34     });
 35     
 36     /**
 37      * Sets wherether text of the view can be selected.
 38      *
 39      * @param {boolean=} state 
 40      * @returns {boolean|uki.view.Base} current textSelectable state of self
 41      */
 42     proto.textSelectable = function(state) {
 43         if (state === undefined) return this._textSelectable;
 44         
 45         this._textSelectable = state;
 46         if (this._textSelectProp) {
 47             this._dom.style[this._textSelectProp] = state ? '' : this._textSelectProp == 'MozUserSelect' ? '-moz-none' : 'none';
 48         } else {
 49             this._dom.unselectable = state ? '' : 'on';
 50         }
 51         this._dom.style.cursor = state ? 'text' : 'default';
 52         return this;
 53     };
 54 };
