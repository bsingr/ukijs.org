1 include('const.js');
  2 include('utils.js');
  3 include('dom.js');
  4 include('geometry.js');
  5 include('view/observable.js');
  6 
  7 (function() {
  8     
  9     var self = uki.Attachment = uki.newClass(uki.view.Observable, /** @lends uki.Attachment.prototype */ {
 10         /**
 11          * Attachment serves as a connection between a uki view and a dom container.
 12          * It notifies its view with parentResized on window resize. 
 13          * Attachment supports part of uki.view.Base API like #domForChild or #rectForChild
 14          *
 15          * @param {Element} dom Container element
 16          * @param {uki.view.Base} view Attached view
 17          * @param {uki.geometry.Rect} rect Initial size
 18          *
 19          * @see uki.view.Base#parentResized
 20          * @name uki.Attachment
 21          * @base uki.view.Observable
 22          * @constructor
 23          */
 24         init: function( dom, view, rect ) {
 25             uki.initNativeLayout();
 26             
 27             this._dom     = dom = dom || root;
 28             this._view    = view;
 29             this._rect = Rect.create(rect) || this.rect();
 30             
 31             uki.dom.offset.initialize();
 32             
 33             view.parent(this);
 34             
 35             if (dom != root && dom.tagName != 'BODY') {
 36                 var computedStyle = dom.runtimeStyle || dom.ownerDocument.defaultView.getComputedStyle(dom, null);
 37                 if (!computedStyle.position || computedStyle.position == 'static') dom.style.position = 'relative';
 38             }
 39             self.register(this);
 40 
 41             this.layout();
 42         },
 43         
 44         /**
 45          * Returns document.body if attached to window. Otherwise returns dom
 46          * uki.view.Base api
 47          *
 48          * @type Element
 49          */
 50         domForChild: function() {
 51             return this._dom === root ? doc.body : this._dom;
 52         },
 53         
 54         /**
 55          * uki.view.Base api
 56          *
 57          * @type uki.geometry.Rect
 58          */
 59         rectForChild: function(child) {
 60             return this.rect();
 61         },
 62         
 63         /**
 64          * uki.view.Base api
 65          */
 66         scroll: function() {
 67             // TODO: support window scrolling
 68         },  
 69         
 70         /**
 71          * uki.view.Base api
 72          */
 73         scrollTop: function() {
 74             // TODO: support window scrolling
 75             return this._dom.scrollTop || 0;
 76         },
 77         
 78         /**
 79          * uki.view.Base api
 80          */
 81         scrollLeft: function() {
 82             // TODO: support window scrolling
 83             return this._dom.scrollLeft || 0;
 84         },
 85         
 86         /**
 87          * uki.view.Base api
 88          */
 89         parent: function() {
 90             return null;
 91         },
 92         
 93         /**
 94          * On window resize resizes and laysout its child view
 95          */
 96         layout: function() {
 97             var oldRect = this._rect;
 98                 
 99             // if (rect.eq(this._rect)) return;
100             this._rect = this.rect();
101 
102             this._view.parentResized(oldRect, this._rect);
103             if (this._view._needsLayout) this._view.layout();
104             this.trigger('layout', {source: this, rect: this._rect});
105         },
106         
107         /**
108          * @return {Element} Container dom
109          */
110         dom: function() {
111             return this._dom;
112         },
113         
114         /**
115          * @return {Element} Child view
116          */
117         view: function() {
118             return this._view;
119         },
120         
121         /**
122          * @private
123          * @return {uki.geometry.Rect} Size of the container
124          */
125         rect: function() {
126             var width = this._dom === root || this._dom === doc.body ? MAX(getRootElement().clientWidth, this._dom.offsetWidth || 0) : this._dom.offsetWidth,
127                 height = this._dom === root || this._dom === doc.body ? MAX(getRootElement().clientHeight, this._dom.offsetHeight || 0) : this._dom.offsetHeight;
128             
129             return new Rect(width, height);
130         }
131     });
132     
133     function getRootElement() {
134         return doc.compatMode == "CSS1Compat" && doc.documentElement || doc.body;
135     }
136     
137     self.instances = [];
138     
139     self.register = function(a) {
140         if (self.instances.length == 0) {
141             uki.dom.bind(root, 'resize', function() {
142                 uki.each(self.instances, function() { this.layout(); });
143             });
144         }
145         self.instances.push(a);
146     };
147     
148     self.childViews = function() {
149         return uki.map(self.instances, 'view');
150     };
151     
152     uki.top = function() {
153         return [self];
154     };
155 })();
