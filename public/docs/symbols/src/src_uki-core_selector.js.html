  5 (function() {
  6     var self,
  7         marked = '__selector_marked',
  8         attr = uki.attr,
  9         chunker = /((?:\((?:\([^()]+\)|[^()]+)+\)|\[(?:\[[^[\]]*\]|['"][^'"]*['"]|[^[\]'"]+)+\]|\\.|[^ >+~,(\[\\]+)+|[>+~])(\s*,\s*)?/g,
 10         regexps = [ // enforce order
 11     		{ name: 'ID', regexp: /#((?:[\w\u00c0-\uFFFF_-]|\\.)+)/ },
 12     		{ name: 'ATTR', regexp: /\[\s*((?:[\w\u00c0-\uFFFF_-]|\\.)+)\s*(?:(\S?=)\s*(['"]*)(.*?)\3|)\s*\]/ },
 13     		{ name: 'TYPE', regexp: /^((?:[\w\u00c0-\uFFFF\*_\.-]|\\.)+)/ },
 14     		{ name: 'POS',  regexp: /:(nth|eq|gt|lt|first|last|even|odd)(?:\((\d*)\))?(?=[^-]|$)/ }
 15 	    ],
 16 	    posRegexp = regexps.POS,
 17 	    posFilters = {
 18     		first: function(i){
 19     			return i === 0;
 20     		},
 21     		last: function(i, match, array){
 22     			return i === array.length - 1;
 23     		},
 24     		even: function(i){
 25     			return i % 2 === 0;
 26     		},
 27     		odd: function(i){
 28     			return i % 2 === 1;
 29     		},
 30     		lt: function(i, match){
 31     			return i < match[2] - 0;
 32     		},
 33     		gt: function(i, match){
 34     			return i > match[2] - 0;
 35     		},
 36     		nth: function(i, match){
 37     			return match[2] - 0 == i;
 38     		},
 39     		eq: function(i, match){
 40     			return match[2] - 0 == i;
 41     		}
 42     	},
 43 	    reducers = {
 44     		TYPE: function(comp, match) {
 45     		    var expected = match[1];
 46     		    if (expected == '*') return true;
 47     		    var typeName = attr(comp, 'typeName');
 48     		    return typeName && typeName.length >= expected.length && 
 49     		           ('.' + typeName).indexOf('.' + expected) == (typeName.length - expected.length);
 50     		},
 51     		
 52     		ATTR: function(comp, match) {
 53     			var result = attr(comp, match[1]),
 54     			    value = result + '',
 55     				type = match[2],
 56     				check = match[4];
 57     				
 58                 return result == null ? type === "!=" :
 59                        type === "="   ? value === check :
 60                        type === "*="  ? value.indexOf(check) >= 0 :
 61                        type === "~="  ? (" " + value + " ").indexOf(check) >= 0 :
 62                        !check         ? value && result !== false :
 63                        type === "!="  ? value != check :
 64                        type === "^="  ? value.indexOf(check) === 0 :
 65                        type === "$="  ? value.substr(value.length - check.length) === check :
 66                        false;
 67     		},
 68     		
 69     		ID: function(comp, match) {
 70     		    return reducers.ATTR(comp, ['', 'id', '=', '', match[1]]);
 71     		},
 72     		
 73     		POS: function(comp, match, i, array) {
 74     			var filter = posFilters[match[1]];
 75 				return filter ? filter( i, match, array ) : false;
 76     		}
 77 	    },
 78 	    mappers = {
 79     		"+": function(context){
 80     		},
 81     		
 82     		">": function(context){
 83     		    return flatten(uki.map(context, 'childViews'));
 84     		},
 85     		
 86     		"": function(context) {
 87     		    return recChildren(flatten(uki.map(context, 'childViews')));
 88     		},
 89     		
 90     		"~": function(context){
 91     		}	        
 92 	    };
 93 	
 94 	function recChildren (comps) {
 95 	    return flatten(uki.map(comps, function(comp) {
 96 	        return [comp].concat( recChildren(attr(comp, 'childViews')) );
 97 	    }));
 98 	}
 99 	    
100 	function flatten (array) {
101 	   return uki.reduce( [], array, reduceFlatten );
102 	}
103 	    
104 	function reduceFlatten (x, e) {
105 	   return x.concat(e);
106 	}
107 	
108 	function removeDuplicates (array) {
109 	    var result = [],
110 	        i;
111 	        
112 	    for (i = 0; i < array.length; i++) { 
113 	        if (!array[i][marked]) { result[result.length] = array[i]; }
114 	        array[i][marked] = true;
115 	    };
116 	    for (i = 0; i < result.length; i++) { result[i][marked] = undefined; };
117 	    return result;
118 	}

121     /**
122      * @namespace
123      */
124     uki.Selector = {
125     	/**
126     	 * <p>Finds views by CSS3 selectors in view tree.</p>
127     	 * <p>Can be called as uki(selector) instead of uki.Selector.find(selector)</p>
128     	 *
129     	 * @example
130     	 *   uki('Label') find all labels on page
131     	 *   uki('Box[name=main] > Label') find all imidiate descendant Labels in a box with name = "main"
132     	 *   uki('> Slider', context) find all direct descendant Sliders within given context
133     	 *   uki('Slider,Checkbox') find all Sliders and Checkboxes
134     	 *   uki('Slider:eq(3)') find 3-d slider
135     	 *
136     	 * @param {string} selector
137     	 * @param {Array.<uki.view.Base>} context to search in
138     	 *
139     	 * @return {uki.Collection} found views
140     	 */    
141         find: function(selector, context, skipFiltering) {
142             context = context || uki.top();
143             if (context.length === undefined) context = [context];
144 
145             var tokens = self.tokenize(selector),
146                 expr   = tokens[0],
147                 extra  = tokens[1],
148                 result = context,
149                 mapper;
150                 
151             while (expr.length > 0) {
152                 mapper = mappers[expr[0]] ? mappers[expr.shift()] : mappers[''];
153                 result = mapper(result);
154                 if (expr.length == 0) break;
155                 result = self.reduce(expr.shift(), result);
156             }
157 
158             if (extra) {
159                 result = result.concat(self.find(extra, context, true));
160             }
161             
162             return skipFiltering ? result : new uki.Collection(removeDuplicates(result));
163         },
164         
165         /** @ignore */
166         reduce: function(exprItem, context) {
167             if (!context || !context.length) return [];
168             
169             var match, found;
170                 
171             while (exprItem != '') {
172                 found = false;
173                 uki.each(regexps, function(index, row) {
174                     
175                     /*jsl:ignore*/
176                     if (match = exprItem.match(row.regexp)) {
177                     /*jsl:end*/
178                         found = true;
179                         context = uki.grep(context, function(comp, index) { 
180                             return reducers[row.name](comp, match, index, context); 
181                         });
182                         exprItem = exprItem.replace(row.regexp, '');
183                         return false;
184                     }
185                 });
186                 if (!found) break;
187             }
188             return context;
189         },
190         
191         /** @ignore */
192         tokenize: function(expr) {
193         	var parts = [], match, extra;
194 
195         	chunker.lastIndex = 0;
196 
197         	while ( (match = chunker.exec(expr)) !== null ) {
198         		parts.push( match[1] );
199 
200         		if ( match[2] ) {
201         			extra = RegExp.rightContext;
202         			break;
203         		}
204         	}
205             
206             return [parts, extra];
207         }
208     };
