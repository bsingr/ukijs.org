1 uki.extend(uki.dom, /** @lends uki.dom */ {
  2     bound: {},
  3     handles: {},
  4     
  5     events: ("blur,focus,load,resize,scroll,unload,click,dblclick," +
  6     	"mousedown,mouseup,mousemove,mouseover,mouseout,mouseenter,mouseleave," +
  7     	"change,select,submit,keydown,keypress,keyup,error").split(","),
  8 
  9     bind: function(el, types, handler) {
 10 		if ( el.setInterval && el != window )
 11 			el = window;
 12 			
 13         handler.huid = handler.huid || uki.dom.guid++;
 14         
 15         var id = el[expando] = el[expando] || uki.dom.guid++,
 16             handle = uki.dom.handles[id] = uki.dom.handles[id] || function() {
 17                 uki.dom.handler.apply(arguments.callee.elem, arguments);
 18             },
 19             i, type;
 20             
 21         handle.elem = el;
 22         
 23         if (!uki.dom.bound[id]) uki.dom.bound[id] = {};
 24         
 25         types = types.split(' ');
 26         for (i=0; i < types.length; i++) {
 27             type = types[i];
 28             if (!uki.dom.bound[id][type]) {
 29                 el.addEventListener ? el.addEventListener(type, handle, false) : el.attachEvent('on' + type, handle);
 30                 uki.dom.bound[id][type] = [];
 31             }
 32             uki.dom.bound[id][type].push(handler);
 33         };
 34         handler = handle = el = null;
 35     },
 36     
 37     unbind: function(el, types, handler) {
 38         var id = el[expando],
 39             huid = handler.huid,
 40             i, type;
 41         types = types.split(' ');
 42         for (i=0; i < types.length; i++) {
 43             type = types[i];
 44             if (!huid || !id || !uki.dom.bound[id] || !uki.dom.bound[id][type]) continue;
 45             uki.dom.bound[id][type] = uki.grep(uki.dom.bound[id][type], function(h) { return h.huid !== huid; });
 46         }
 47     },
 48     
 49     /** @ignore */
 50     handler: function( e ) {
 51         e = uki.dom.fix( e || root.event );
 52 
 53         var type = e.type,
 54             id = this[expando],
 55             handlers = uki.dom.bound[id],
 56             i;
 57             
 58         if (!id || !handlers || !handlers[type]) return;
 59         
 60         for (i=0, handlers = handlers[type]; i < handlers.length; i++) {
 61             handlers[i].call(this, e);
 62         };
 63     },
 64     
 65     preventDefault: function(e) {
 66         if (!e) return;
 67         e.preventDefault ? e.preventDefault() : e.returnValue = false;
 68     },
 69     
 70     /**
 71      * Taken from jQuery
 72      * @ignore
 73      */
 74     fix: function( event ) {
 75 		// Fix target property, if necessary
 76 		if ( !event.target )
 77 			event.target = event.srcElement || doc;
 78 
 79 		// check if target is a textnode (safari)
 80 		if ( event.target.nodeType == 3 )
 81 			event.target = event.target.parentNode;
 82 
 83 		// Add relatedTarget, if necessary
 84 		if ( !event.relatedTarget && event.fromElement )
 85 			event.relatedTarget = event.fromElement == event.target ? event.toElement : event.fromElement;
 86 
 87 		// Calculate pageX/Y if missing and clientX/Y available
 88 		if ( event.pageX == null && event.clientX != null ) {
 89 			var de = doc.documentElement, body = doc.body;
 90 			event.pageX = event.clientX + (de && de.scrollLeft || body && body.scrollLeft || 0) - (de.clientLeft || 0);
 91 			event.pageY = event.clientY + (de && de.scrollTop  || body && body.scrollTop || 0)  - (de.clientTop || 0);
 92 		}
 93 
 94 		// Add which for key events
 95 		if ( !event.which && ((event.charCode || event.charCode === 0) ? event.charCode : event.keyCode) )
 96 			event.which = event.charCode || event.keyCode;
 97 
 98 		// Add metaKey to non-Mac browsers (use ctrl for PC's and Meta for Macs)
 99 		if ( !event.metaKey && event.ctrlKey )
100 			try { event.metaKey = event.ctrlKey; } catch(e){};
101 
102 		// Add which for click: 1 == left; 2 == middle; 3 == right
103 		// Note: button is not normalized, so don't use it
104 		if ( !event.which && event.button )
105 			event.which = (event.button & 1 ? 1 : ( event.button & 2 ? 3 : ( event.button & 4 ? 2 : 0 ) ));    
106 			
107 		return event;    
108     }
109 });

111 if (root.attachEvent) {
112     root.attachEvent('onunload', function() {
113         uki.each(uki.dom.bound, function(id, types) {
114             uki.each(types, function(type, handlers) {
115                 try {
116                     uki.dom.handles[id].elem.detachEvent('on' + type, uki.dom.handles[id]);
117                 } catch (e) {};
118             });
119         });
120     });
