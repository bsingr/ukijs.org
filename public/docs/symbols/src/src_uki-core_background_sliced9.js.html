  4 /**
  5  * Adds a div with 9 sliced images in the corners, sides and center
  6  *
  7  * @class
  8  */
  9 uki.background.Sliced9 = uki.newClass(new function() {
 10     var nativeCss = ['MozBorderImage', 'WebkitBorderImage', 'borderImage'],
 11         dom = uki.dom;
 12         
 13     var LEFT = 'left:',
 14         TOP  = 'top:',
 15         RIGHT = 'right:',
 16         BOTTOM = 'bottom:',
 17         WIDTH = 'width:',
 18         HEIGHT = 'height:',
 19         PX = 'px',
 20         P100 = '100%';
 21         
 22     var cache = {},
 23         proto = this;
 24     
 25     proto.init = function(partSettings, inset, options) {
 26         this._settings  = uki.extend({}, partSettings);
 27         this._inset     = Inset.create(inset);
 28         this._size      = null;
 29         this._inited    = false;
 30         
 31         options = options || {};
 32         this._fixedSize = Size.create(options.fixedSize) || new Size();
 33         this._bgInset   = Inset.create(options.inset)    || new Inset();
 34         this._zIndex    = options.zIndex || -1;
 35 
 36         this._container = this._getContainer();
 37         this._container.style.zIndex = this._zIndex;
 38     };
 39     
 40     function makeDiv (name, style, setting, imgStyle, bgSettings) {
 41         var inner = setting[3] ? img(setting, imgStyle) : '';
 42         if (!setting[3]) style += bgStyle(setting, bgSettings);
 43         return '<div class="' +  name + '" style="position:absolute;overflow:hidden;' + style + '">' + inner + '</div>';
 44     }
 45     
 46     function bgStyle (setting, bgSettings) {
 47         return ';background: url(' + uki.imageSrc(setting[0], setting[1], setting[2]) + ') ' + bgSettings;
 48     }
 49 
 50     function img (setting, style) {
 51         return uki.imageHTML(setting[0], setting[1], setting[2], ' galleryimg="no" style="-webkit-user-drag:none;position:absolute;' + style + '"');
 52     }
 53     
 54     proto._getContainer = function() {
 55         var key = this._getKey();
 56         if (!cache[key]) {
 57             return cache[key] = this._createContainer();
 58         }
 59         return cache[key].cloneNode(true);
 60     };
 61     
 62     proto._createContainer = function() {
 63         var inset = this._inset,
 64             bgInset = this._bgInset,
 65             settings = this._settings,
 66             width = inset.left + inset.right,
 67             height = inset.top + inset.bottom,
 68             css = [LEFT + bgInset.left + PX, RIGHT + bgInset.right + PX, TOP + bgInset.top + PX, BOTTOM + bgInset.bottom + PX].join(';'),
 69             html = [];
 70             
 71         if (inset.top && inset.left) {
 72             html[html.length] = makeDiv('tl',
 73                 [LEFT + 0, TOP + 0, WIDTH + inset.left + PX, HEIGHT + inset.top + PX].join(';'),
 74                 settings.c, [LEFT + 0, TOP + 0, WIDTH + width + PX, HEIGHT + height + PX].join(';'), 'top left'
 75             );
 76         }
 77         if (inset.top) {
 78             html[html.length] = makeDiv('t',
 79                 [LEFT + inset.left + PX, TOP + 0, HEIGHT + inset.top + PX, RIGHT + inset.right + PX].join(';'),
 80                 settings.h, [LEFT + 0, TOP + 0, WIDTH + P100, HEIGHT + height + PX].join(';'), 'repeat-x top'
 81             );
 82         }
 83         if (inset.top && inset.right) {
 84             html[html.length] = makeDiv('tr',
 85                 [RIGHT + 0, TOP + 0, WIDTH + inset.right + PX, HEIGHT + inset.top + PX].join(';'),
 86                 settings.c, [LEFT + '-' + inset.left + PX, TOP + 0, WIDTH + width + PX, HEIGHT + height + PX].join(';'), 'top right'
 87             );
 88         }
 89         
 90         if (inset.left) {
 91             html[html.length] = makeDiv('l',
 92                 [LEFT + 0, TOP + inset.top + PX, WIDTH + inset.left + PX, BOTTOM + inset.bottom + PX].join(';'),
 93                 settings.v, [LEFT + 0, TOP + 0, HEIGHT + P100, WIDTH + width + PX].join(';'), 'repeat-y left'
 94             );
 95         }
 96         if (settings.m) {
 97             html[html.length] = makeDiv('m',
 98                 [LEFT + inset.left + PX, TOP + inset.top + PX, RIGHT + inset.left + PX, BOTTOM + inset.bottom + PX].join(';'),
 99                 settings.m, [LEFT + 0, TOP + 0, HEIGHT + P100, WIDTH + P100].join(';'), ''
100             );
101         }
102         if (inset.right) {
103             html[html.length] = makeDiv('r',
104                 [RIGHT + 0, TOP + inset.top + PX, WIDTH + inset.right + PX, BOTTOM + inset.bottom + PX].join(';'),
105                 settings.v, [LEFT + '-' + inset.left + PX, TOP + 0, HEIGHT + P100, WIDTH + width + PX].join(';'), 'repeat-y right'
106             );
107         }
108         
109         if (inset.bottom && inset.left) {
110             html[html.length] = makeDiv('bl',
111                 [LEFT + 0, BOTTOM + 0, WIDTH + inset.left + PX, HEIGHT + inset.bottom + PX].join(';'),
112                 settings.c, [LEFT + 0, TOP + '-' + inset.top + PX, WIDTH + width + PX, HEIGHT + height + PX].join(';'), 'left -' + inset.top + PX
113             );
114         }
115         if (inset.bottom) {
116             html[html.length] = makeDiv('b',
117                 [LEFT + inset.left + PX, BOTTOM + 0, HEIGHT + inset.bottom + PX, RIGHT + inset.right + PX].join(';'),
118                 settings.h, [LEFT + 0, TOP + '-' + inset.top + PX, WIDTH + P100, HEIGHT + height + PX].join(';'), 'repeat-x 0 -' + inset.top + PX
119             );
120         }
121         if (inset.bottom && inset.right) {
122             html[html.length] = makeDiv('br',
123                 [RIGHT + 0, BOTTOM + 0, WIDTH + inset.right + PX, HEIGHT + inset.bottom + PX].join(';'),
124                 settings.c, [LEFT + '-' + inset.left + PX, TOP + '-' + inset.top + PX, WIDTH + width + PX, HEIGHT + height + PX].join(';'), 'right -' + inset.top + PX
125             );
126         }
127         return uki.createElement('div', 'position:absolute;overflow:hidden;' + css, html.join(''));
128     };
129     
130     proto._getKey = function() {
131         return uki.map(['v', 'h', 'm', 'c'], function(x) {
132             return this._settings[x] && this._settings[x][0] || '';
133         }, this).concat([this._inset, this._bgInset, this._fixedSize]).join(',');
134     };
135     
136     proto.attachTo = function(comp) {
137         this._comp = comp;
138         
139         this._container.style.visibility = 'visible';
140         this._comp.dom().appendChild(this._container);
141         
142         if (!uki.supportNativeLayout) {
143             var _this = this;
144             this._layoutHandler = this._layoutHandler || function(e) {
145                 if (_this._size && _this._size.eq(e.rect)) return;
146                 _this._size = e.rect;
147                 _this.layout();
148             };
149             this._comp.bind('layout', this._layoutHandler);
150             this.layout();
151         }
152     };
153     
154     proto.detach = function() {
155         if (this._comp) {
156             // this._comp.dom().removeChild(this._container);
157             this._container.style.visibility = 'hidden';
158             if (!uki.supportNativeLayout) this._comp.unbind('layout', this._layoutHandler);
159             this._size = this._comp = null;
160             this._attached = this._inited = false;
161         }
162     };
163     
164     proto.layout = function(e) {
165         var size = this._comp.rect(),
166             parts = this._parts,
167             inset = this._inset,
168             bgInset = this._bgInset,
169             fixedSize = this._fixedSize,
170             width = FLOOR(fixedSize.width || size.width - bgInset.left - bgInset.right),
171             height = FLOOR(fixedSize.height || size.height - bgInset.top - bgInset.bottom),
172             insetWidth = inset.left + inset.right,
173             insetHeight = inset.top + inset.bottom;
174             
175         if (!parts) {
176             parts = {};
177             uki.each(this._container.childNodes, function() {
178                 if (this.className) parts[this.className] = this;
179             });
180             this._parts = parts;
181         }
182         // parts.b.style.bottom = ''
183         // parts.b.style.top = '100%';
184         // parts.b.style.marginTop = - inset.bottom + 'px';
185         if (parts.t) dom.layout(parts.t.style, { width: width - insetWidth });
186         if (parts.b) dom.layout(parts.b.style, { width: width - insetWidth });
187         if (parts.l) dom.layout(parts.l.style, { height: height - insetHeight });
188         if (parts.r) dom.layout(parts.r.style, { height: height - insetHeight });
189         if (parts.m) dom.layout(parts.m.style, {
190             height: height - insetHeight,
191             width: width - insetWidth
192         });
193         dom.layout(this._container.style, {
194             width: width,
195             height: height
196         });
197     };
198 });
