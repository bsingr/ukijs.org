  5 (function() {
  6     
  7     /**
  8      * <p>Collection performs group operations on uki.view objects.</p>
  9      * <p>Behaves much like result jQuery(dom nodes).
 10      * Most methods are chainable like .attr('text', 'somevalue').bind('click', function() { ... })</p>
 11      *
 12      * <p>Its easier to call uki([view1, view2]) or uki('selector') instead of creating collection directly</p>
 13      *
 14      * @author voloko
 15      * @constructor
 16      * @class
 17      */
 18     uki.Collection = function( elems ) {
 19         this.length = 0;
 20     	Array[PROTOTYPE].push.apply( this, elems );
 21     };
 22 
 23     /** @type uki.Collection.prototype */
 24     var self = uki.fn = uki.Collection[PROTOTYPE] = {};
 25     
 26     /**
 27      * Iterates trough all items within itself
 28      *
 29      * @name uki.Collection#each
 30      * @function
 31      *
 32      * @param {function(this:uki.view.Base, number, uki.view.Base)} callback Callback to call for every item
 33      * @returns {uki.view.Collection} self
 34      */
 35     self.each = function( callback ) {
 36         uki.each( this, callback );
 37         return this;
 38     };
 39     
 40     /**
 41      * Creates a new uki.Collection populated with found items
 42      *
 43      * @name uki.Collection#grep
 44      * @function
 45      *
 46      * @param {function(uki.view.Base, number):boolean} callback Callback to call for every item
 47      * @returns {uki.view.Collection} created collection
 48      */
 49     self.grep = function( callback ) {
 50         return new uki.Collection( uki.grep(this, callback) );
 51     };
 52     
 53     /**
 54      * Sets an attribute on all views or gets the value of the attribute on the first view
 55      *
 56      * @example
 57      * c.attr('text', 'my text') // sets text to 'my text' on all collection views
 58      * c.attr('name') // gets name attribute on the first view
 59      *
 60      * @name uki.Collection#attr
 61      * @function
 62      *
 63      * @param {string} name Name of the attribute
 64      * @param {object=} value Value to set
 65      * @returns {uki.view.Collection|Object} Self or attribute value
 66      */
 67     self.attr = function( name, value ) {
 68         if (value !== undefined) {
 69             this.each(function() {
 70                 uki.attr( this, name, value );
 71             });
 72             return this;
 73         } else {
 74             return this[0] ? uki.attr( this[0], name ) : '';
 75         }
 76     };
 77     
 78     /**
 79      * Finds views within collection context
 80      * @example
 81      * c.find('Button')
 82      *
 83      * @name uki.Collection#find
 84      * @function
 85      *
 86      * @param {string} selector 
 87      * @returns {uki.view.Collection} Collection of found items
 88      */
 89     self.find = function( selector ) {
 90         return uki.find( selector, this );
 91     };
 92     
 93     /**
 94      * Attaches all child views to dom container
 95      *
 96      * @name uki.Collection#attachTo
 97      * @function
 98      *
 99      * @param {Element} dom Container dom element
100      * @param {uki.geometry.Rect} rect Default size
101      * @returns {uki.view.Collection} self
102      */
103     self.attachTo = function( dom, rect ) {
104         this.each(function() {
105             new uki.Attachment( dom, this, rect );
106         });
107         return this;
108     };
109     
110     /**
111      * Appends views to the first item in collection
112      *
113      * @name uki.Collection#append
114      * @function
115      *
116      * @param {Array.<uki.view.Base>} views Views to append
117      * @returns {uki.view.Collection} self
118      */
119     self.append = function( views ) {
120         if (!this[0]) return this;
121         views = views.length !== undefined ? views : [views];
122         for (var i=0; i < views.length; i++) {
123             this[0].appendChild(views[i]);
124         };
125         return this;
126     };
127     
128     uki.each(['parent'], function(i, name) {
129         self[name] = function() {
130             return new uki.Collection( uki.map(this, name) );
131         };
132     });
133     
134     uki.each(['scrollableParent'], function(i, name) {
135        self[name] = function() {
136             return new uki.Collection( uki.map(this, function(c) { return uki.view[name](c); }) );
137         };
138     });
139     
140     uki.each(('bind,unload,trigger,layout,appendChild,removeChild,insertBefore,addRow,removeRow,' +
141         'resizeToContents,toggle').split(','), function(i, name) {
142         self[name] = function() { 
143             for (var i=0; i < this.length; i++) {
144                 this[i][name].apply(this[i], arguments);
145             };
146             return this;
147         };
148     });
149     
150     uki.each(('html,text,background,value,rect,checked,selectedIndex,' +
151         'childViews,typeName,id,name,visible,disabled,focusable').split(','), function(i, name) {
152         self[name] = function( value ) { return this.attr( name, value ); };
153     });
154     
155     uki.each( ("blur,focus,load,resize,scroll,unload,click,dblclick," +
156     	"mousedown,mouseup,mousemove,mouseover,mouseout,mouseenter,mouseleave," +
157     	"change,select,submit,keydown,keypress,keyup,error").split(","), function(i, name){
158     	self[name] = function( handler ){
159     	    if (handler) {
160         		this.bind(name, handler);
161     	    } else {
162                 for (var i=0; i < this.length; i++) {
163                     this[i][name]();
164                 };
165     	    }
166     		return this;
167     	};
168     });
169 })();
