  5 /**
  6  * Basic utils to work with the dom tree
  7  * @namespace
  8  * @author voloko
  9  */
 10 uki.dom = {
 11     guid: 1,
 12     
 13     /**
 14      * Convinience wrapper around document.createElement
 15      * Creates dom element with given tagName, cssText and innerHTML
 16      *
 17      * @param {string} tagName
 18      * @param {string=} cssText
 19      * @param {string=} innerHTML
 20      * @returns {Element} created element
 21      */
 22     createElement: function(tagName, cssText, innerHTML) {
 23         var e = doc.createElement(tagName);            
 24         if (cssText) e.style.cssText = cssText;
 25         if (innerHTML) e.innerHTML = innerHTML;
 26         e[expando] = uki.dom.guid++;
 27         return e;
 28     },
 29     
 30     /**
 31      * Adds a probe element to page dom tree, callbacks, removes the element
 32      *
 33      * @param {Element} dom Probing dom element
 34      * @param {function(Element)} callback
 35      */
 36     probe: function(dom, callback) {
 37         var target = doc.body;
 38         target.appendChild(dom);
 39         callback(dom);
 40         target.removeChild(dom);
 41     },
 42     
 43     /**
 44      * Assigns layout style properties to an element
 45      *
 46      * @param {CSSStyleDeclaration} style Target declaration
 47      * @param {object} layout Properties to assign
 48      * @param {object=} prevLayout If given assigns only differenct between layout and prevLayout
 49      */
 50     layout: function(style, layout, prevLayout) {
 51         prevLayout = prevLayout || {};
 52         if (prevLayout.left   != layout.left)   style.left   = layout.left + PX;
 53         if (prevLayout.top    != layout.top)    style.top    = layout.top + PX;
 54         if (prevLayout.right  != layout.right)  style.right  = layout.right + PX;
 55         if (prevLayout.bottom != layout.bottom) style.bottom = layout.bottom + PX;
 56         if (prevLayout.width  != layout.width)  style.width  = MAX(layout.width, 0) + PX;
 57         if (prevLayout.height != layout.height) style.height = MAX(layout.height, 0) + PX;
 58         return layout;
 59     },
 60     
 61     /**
 62      * Computed style for a give element
 63      *
 64      * @param {Element} el
 65      * @returns {CSSStyleDeclaration} style declaratioin
 66      */
 67     computedStyle: function(el) {
 68         if (doc && doc.defaultView && doc.defaultView.getComputedStyle) {
 69             return doc.defaultView.getComputedStyle( el, null );
 70         } else if (el.currentStyle) {
 71             return el.currentStyle;
 72         }
 73     },
 74     
 75     /**
 76      * Checks if parent contains child
 77      *
 78      * @param {Element} parent 
 79      * @param {Element} child 
 80      * @return {Boolean}
 81      */
 82     contains: function(parent, child) {
 83         try {
 84             if (parent.contains) return parent.contains(child);
 85             if (parent.compareDocumentPosition) return parent.compareDocumentPosition(child) & 16;
 86         } catch (e) {}
 87         while ( child && child != parent ) {
 88             try { child = child.parentNode } catch(e) { child = null };
 89         }
 90         return parent == child;
 91     },
 92     
 93     createStylesheet: function(code) {
 94         var style = doc.createElement('style');
 95         doc.getElementsByTagName('head')[0].appendChild(style);
 96         if (style.styleSheet) { //IE
 97             style.styleSheet.cssText = code;
 98         } else {
 99             style.appendChild(document.createTextNode(code));
100         }
101         return style;
102     }
103     
104 };
