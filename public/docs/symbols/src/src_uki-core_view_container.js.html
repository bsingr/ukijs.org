 25         var val = 0, i, l;
 26         for (i = c._childViews.length - 1; i >= 0; i--){
 27             val = MAX(val, c._childViews[i].rect()[prop]());

 30     }

 38     };

 44     /**
 45      * Sets or retrievs view child view.
 46      * @param anything uki.build can parse
 47      *
 48      * Note: if setting on view with child views, all child view will be removed
 49      */
 50     proto.childViews = function(val) {
 51         if (val === undefined) return this._childViews;
 52         uki.each(this._childViews, function(i, child) {
 53             this.removeChild(child);
 54         }, this);
 55         uki.each(uki.build(val), function(tmp, child) {
 56             this.appendChild(child);
 57         }, this);
 58         return this;
 59     };
 60     
 61     /**
 62      * Remove particular child
 63      */
 64     proto.removeChild = function(child) {
 65         child.parent(null);
 66         this.domForChild(child).removeChild(child.dom());
 67         var index = child._viewIndex,
 68             i, l;
 69         for (i=index+1, l = this._childViews.length; i < l; i++) {
 70             this._childViews[i]._viewIndex--;
 71         };
 72         this._childViews = uki.grep(this._childViews, function(elem) { return elem != child; });
 73     };
 74     
 75     /**
 76      * Adds a child.
 77      */
 78     proto.appendChild = function(child) {
 79         child._viewIndex = this._childViews.length;
 80         this._childViews.push(child);
 81         child.parent(this);
 82         this.domForChild(child).appendChild(child.dom());
 83     };
 84     
 85     /**
 86      * Insert child before target beforeChild
 87      * @param {uki.view.Base} child Child to insert
 88      * @param {uki.view.Base} beforeChild Existend child before which we should insert
 89      */
 90     proto.insertBefore = function(child, beforeChild) {
 91         var i, l;
 92         child._viewIndex = beforeChild._viewIndex;
 93         for (i=beforeChild._viewIndex, l = this._childViews.length; i < l; i++) {
 94             this._childViews[i]._viewIndex++;
 95         };
 96         this._childViews.splice(beforeChild._viewIndex, 0, child);
 97         child.parent(this);
 98         this.domForChild(child).insertBefore(child.dom(), beforeChild.dom());
 99     };
100     
101     /**
102      * Should return a dom node for a child.
103      * Child should append itself to this dom node
104      */
105     proto.domForChild = function(child) {

115     proto._layoutChildViews = function() {
116         for (var i=0, childViews = this.childViews(); i < childViews.length; i++) {
117             if (childViews[i]._needsLayout && childViews[i].visible()) {
118                 childViews[i].layout(this._rect);
119             }
120         };
121     };

124     proto.rect = function(newRect) {
125         if (newRect === undefined) return this._rect;
126 
127         newRect = Rect.create(newRect);
128         this._parentRect = newRect;
129         var oldRect = this._rect;
130         if (!this._resizeSelf(newRect)) return this;
131         this._needsLayout = true;
132         
133         if (oldRect.width != newRect.width || oldRect.height != newRect.height) this._resizeChildViews(oldRect);
134         this.trigger('resize', {oldRect: oldRect, newRect: this._rect, source: this});
135         return this;
136     };
137     
138     proto._resizeSelf = function(newRect) {
139         // if (newRect.eq(this._rect)) return false;
140         this._rect = this._normalizeRect(newRect);
141         return true;
142     };
143     
144     /**
145      * Called to notify all intersted parties: childViews and observers
146      */
147     proto._resizeChildViews = function(oldRect) {
148         for (var i=0, childViews = this.childViews(); i < childViews.length; i++) {
149             childViews[i].parentResized(oldRect, this._rect);
150         };
151     };
