  4 /**
  5  * Creates image element from url
  6  *
  7  * @param {string} url Image url
  8  * @param {string=} dataUrl Data url representation of image, used if supported
  9  * @param {string=} alphaUrl Gif image url for IE6
 10  *
 11  * @namespace
 12  *
 13  * @returns {Element}
 14  */
 15 uki.image = function(url, dataUrl, alphaUrl) {
 16     var result = new Image();
 17     result.src = uki.imageSrc(url, dataUrl, alphaUrl);
 18     return result;
 19 };
 20 
 21 /**
 22  * Selects image src depending on browser
 23  *
 24  * @param {string} url Image url
 25  * @param {string=} dataUrl Data url representation of image, used if supported
 26  * @param {string=} alphaUrl Gif image url for IE6
 27  *
 28  * @returns {string}
 29  */
 30 uki.imageSrc = function(url, dataUrl, alphaUrl) {
 31     if (uki.image.dataUrlSupported && dataUrl) return dataUrl;
 32     if (alphaUrl && uki.image.needAlphaFix) return alphaUrl;
 33     return url;
 34 };
 35 
 36 /**
 37  * Image html representation: '<img src="">'
 38  *
 39  * @param {string} url Image url
 40  * @param {string=} dataUrl Data url representation of image, used if supported
 41  * @param {boolean=} alphaUrl Gif image url for IE6
 42  * @param {string=} html Additional html
 43  *
 44  * @returns {string} html
 45  */
 46 uki.imageHTML = function(url, dataUrl, alphaUrl, html) {
 47     if (uki.image.needAlphaFix && alphaUrl) {
 48         url = alphaUrl;
 49     } else if (uki.image.dataUrlSupported) {
 50         url = dataUrl;
 51     }
 52     return '<img' + (html || '') + ' src="' + url + '" />';
 53 };

 55 /**
 56  * Loads given images, callbacks after all of them loads
 57  *
 58  * @param {Array.<Element>} images Images to load
 59  * @param {function()} callback
 60  */
 61 uki.image.load = function(images, callback) {
 62     var imagesToLoad = images.length;
 63     
 64     uki.each(images, function(i, img) {
 65         if (!img || img.width) {
 66             if (!--imagesToLoad) callback();
 67             return;
 68         }
 69 
 70         var handler = function() {
 71                 img.onload = img.onerror = img.onabort = null; // prevent mem leaks
 72                 if (!--imagesToLoad) callback();
 73             };
 74 		img.onload  = handler;
 75 		img.onerror = handler;
 76 		img.onabort = handler;
 77     });
 78 };

 80 /**
 81  * @type boolean
 82  */
 83 uki.image.dataUrlSupported = doc.createElement('canvas').toDataURL || (/MSIE (8)/).test(ua);
 84 
 85 /**
 86  * @type boolean
 87  */
