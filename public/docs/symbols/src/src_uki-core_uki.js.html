1 /**
  2  * Shortcut access to uki.build, uki.Selector.find and uki.Collection constructor
  3  * uki('#id') is also a shortcut for search by id
  4  *
  5  * @param {String|uki.view.Base|Object|Array.<uki.view.Base>} val
  6  * @param {Array.<uki.view.Base>=} optional context for selector
  7  * @class
  8  * @name uki
  9  * @return {uki.Collection}
 10  */
 11 root.uki = function(val, context) {
 12     if (typeof val == 'string') {
 13         var m = val.match(/^#((?:[\w\u00c0-\uFFFF_-]|\\.)+)$/),
 14             e = m && uki._ids[m[1]];
 15         if (m && !context) {
 16             return new uki.Collection( e ? [e] : [] );
 17         }
 18         return uki.find(val, context);
 19     }
 20     if (val.length === undefined) val = [val];
 21     if (val.length > 0 && uki.isFunction(val[0].typeName)) return new uki.Collection(val);
 22     return uki.build(val);
 23 };
 24 
 25 /**
 26  * @type string
 27  * @field
 28  */
 29 uki.version = '0.0.6';
 30 
 31 /**
 32  * Empty function
 33  * @type function():boolean
 34  */
 35 uki.F = function() { return false; };
 36 uki._ids = {};
 37 
 38 uki.registerId = function(comp) {
 39     uki._ids[uki.attr(comp, 'id')] = comp;
 40 }; 
 41 uki.unregisterId = function(comp) {
 42     uki._ids[uki.attr(comp, 'id')] = undefined;
 43 };
