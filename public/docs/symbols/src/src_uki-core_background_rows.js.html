  3 /**
  4  * Adds a div with colored rows to dom
  5  *
  6  * @class
  7  */
  8 uki.background.Rows = uki.newClass(new function() {
  9     var proto = this,
 10         cache = [],
 11         packSize = 100;
 12     
 13     proto.init = function(height, colors) {
 14         this._height = height || 20;
 15         this._colors = uki.isArray(colors) ? colors : colors.split(' ');
 16         this._packSize = CEIL(packSize/this._colors.length)*this._colors.length;
 17         this._renderedHeight = 0;
 18         this._visibleExt = 200;
 19         if (this._colors.length == 1) this._colors = this._colors.concat(['#FFF']);
 20     };
 21     
 22     proto.attachTo = function(comp) {
 23         this._comp && this.detach();
 24         this._comp = comp;
 25         if (!this._container) {
 26             this._container = uki.createElement(
 27                 'div', 
 28                 'position:absolute;left:0;top:0;width:100%;z-index:-1'
 29             );
 30         }
 31         this._layoutHandler = this._layoutHandler || uki.proxy(function(e) { this.layout(e.rect, e.visibleRect); }, this);
 32         this._comp.dom().appendChild(this._container);
 33         this._comp.bind('layout', this._layoutHandler);
 34     };
 35     
 36     proto.layout = function(rect, visibleRect) {
 37         var height = visibleRect ? visibleRect.height + this._visibleExt*2 : rect.maxY();
 38         while (this._renderedHeight < height) {
 39             var h = packSize * this._height,
 40                 c = uki.createElement('div', 'height:' + h + 'px;overflow:hidden;width:100%;', getPackHTML(this._height, this._colors));
 41             this._renderedHeight += h;
 42             this._container.appendChild(c);
 43         }
 44         if (visibleRect) {
 45             this._container.style.top = CEIL((visibleRect.y - this._visibleExt)/this._height/this._colors.length)*this._height*this._colors.length + 'px';
 46         }
 47     };
 48     
 49     proto.detach = function() {
 50         this._comp.dom().removeChild(this._container);
 51         this._comp.unbind('layout', this._layoutHandler);
 52         this._comp = null;
 53     };
 54     
 55     function getPackHTML (height, colors) {
 56         var key = height + ' ' + colors.join(' '),
 57             rows = [],
 58             html = [],
 59             i, l = colors.length;
 60         if (!cache[key]) {
 61             for (i=0; i < l; i++) {
 62                 rows[i] = ['<div style="height:', height, 'px;width:100%;overflow:hidden;', 
 63                             (colors[i] ? 'background:' + colors[i] : ''),
 64                             '"></div>'].join('');
 65             };
 66             for (i=0; i < packSize; i++) {
 67                 html[i] = rows[i%l];
 68             };
 69             cache[key] = html.join('');
 70         }
 71         return cache[key];
 72     }
 73 });
