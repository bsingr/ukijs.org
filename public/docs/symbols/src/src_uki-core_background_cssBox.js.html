1 include('../background.js');
  2 include('../geometry.js');
  3 
  4 /**
  5  * Adds a div with given cssText to dom()
  6  *
  7  * @class
  8  */
  9 uki.background.CssBox = uki.newClass(new function() {
 10     
 11     var cache = {};
 12     
 13     function getInsets(options) {
 14         if (!cache[options]) {
 15             uki.dom.probe(
 16                 uki.createElement('div', options + ';position:absolute;overflow:hidden;left:-999em;width:10px;height:10px;'), 
 17                 function(c) {
 18                     cache[options] = new Inset(
 19                         c.offsetHeight - 10,
 20                         c.offsetWidth - 10
 21                     );
 22                 }
 23             );
 24         }
 25         return cache[options];
 26     }
 27     
 28     this.init = function(options, ext) {
 29         this._options = options;
 30         ext = ext || {};
 31         this._inset = inset = Inset.create(ext.inset) || new Inset();
 32         this._insetWidth  = getInsets(options).left + inset.left + inset.right;
 33         this._insetHeight = getInsets(options).top + inset.top + inset.bottom;
 34 
 35         this._container = uki.createElement(
 36             'div', 
 37             options + ';position:absolute;overflow:hidden;z-index:' + (ext.zIndex || '-1') + ';' + 
 38             'left:' + inset.left + ';top:' + inset.top + 'px;right:' + inset.right + 'px;bottom:' + inset.bottom + 'px'
 39         );
 40         this._attached = false;
 41     };
 42     
 43     this.attachTo = function(comp) {
 44         this._comp = comp;
 45         this._comp.dom().appendChild(this._container);
 46 
 47         if (uki.supportNativeLayout) return;
 48         
 49         this._layoutHandler = this._layoutHandler || uki.proxy(function(e) { this.layout(e.rect); }, this);
 50         this._comp.bind('layout', this._layoutHandler);
 51         this.layout(this._comp.rect());
 52     };
 53     
 54     this.layout = function(size) {
 55         this._prevLayout = uki.dom.layout(this._container.style, {
 56             width: size.width - this._insetWidth,
 57             height: size.height - this._insetHeight
 58         }, this._prevLayout);
 59     };
 60     
 61     this.detach = function() {
 62         if (this._comp) {
 63             this._comp.dom().removeChild(this._container);
 64             if (!uki.supportNativeLayout) this._comp.unbind('layout', this._layoutHandler);
 65             this._attached = false;
 66         }
 67     };
 68 });
